82 FIRSTSEED: int = 1

84 RUNTYPE: str = "col" # 'col' or 'inj' 

85 NPAIRS: int = 30 # 1-32 particle pairs 

86 EMITTANCE: float = 3.75 # normalized emittance 

87 DIMENSIONS: int = 6 # Phase-Space dimensions 

88 WRITEBINS: int = 500 

89 ENERGY: float = None 

90 GAMMA: float = ( 

91 SetDynamically # calculated from ENERGY, needs to be here so asdict() converts it 

92 ) 

93 

94 def __post_init__(self): 

95 # Check Runtype, Energy and Gamma --- 

96 energy_map = {"inj": 450_000.0, "col": 7000_000.0} 

97 if self.RUNTYPE not in energy_map: 

98 raise ValueError(f"RUNTYPE needs to be one of : {list(energy_map.keys())}") 

99 

100 if self.ENERGY is None: 

101 energy = energy_map[self.RUNTYPE] 

102 LOG.debug(f"Energy for '{self.RUNTYPE}' defaults to {energy}") 

103 self.ENERGY = energy 

104 

105 if self.ENERGY >= 6500_000 and list(energy_map.keys())[1] != self.RUNTYPE: 

106 LOG.warning( 

107 f"Runtype is {self.RUNTYPE}, yet energy is set to {self.ENERGY}. Are you sure?" 

108 ) 

109 

110 self.GAMMA = self.ENERGY / 1000 / PMASS 

111 

112 # Check Seeds --- 

113 if any(getattr(self, key) is None for key in SEED_KEYS): # set by user to None 

114 for key in SEED_KEYS: 

115 setattr(self, key, 0) 

116 

117 # the following checks are limits of SixDesk in 2020 

118 # and might be fixed upstream in the future 

119 if self.AMPMAX < self.AMPMIN: 

120 raise ValueError("Given AMPMAX is smaller than AMPMIN.") 

121 

122 if (self.AMPMAX - self.AMPMIN) % self.AMPSTEP: 

123 raise ValueError("The amplitude range need to be dividable by the amplitude steps!") 

124 

125 if not self.ANGLES % 2: 

126 raise ValueError("The number of angles needs to be an uneven one.") 

127
